keepalived虚拟IP

2023/12/7 常用配置类

1.安装：

yum -y install  keepalived 

yum -y install psmisc #按装kill命令

#killall 不可用，可用安装psmisc包解决 
yum install psmisc  -y

#安装网络工具包，可以显示出当前虚拟IP168对应的真正物理网卡地址(mac地址)。
yum -y install net-tools

2.脚本文件

#vim /etc/keepalived/checknginx.sh
#复制下面内容：
#!/bin/bash
A=`ps -C nginx --no-header |wc -l`
if [ $A -eq 0 ];then
     /usr/local/nginx/sbin/nginx
      if [ `ps -C nginx --no-header |wc -l` -eq 0 ];then
              killall keepalived
      fi
fi

3.master配置

#vim /etc/keepalived/keepalived.conf
#复制下面内容：
global_defs {
   notification_email {
     test@qq.com
   }
   #发送通知邮件时邮件源地址是谁
   notification_email_from test@qq.com   
   smtp_server 127.0.0.1
   #连接smtp连接超时时间
   smtp_connect_timeout 30 
   #机器标识
   router_id LVS_DEVEL  
}
#用来做健康检查的，当时检查失败时会将vrrp_instance的priority减少相应的值。
  vrrp_script chk_http_port {
           #监控脚本
           script "/etc/keepalived/checknginx.sh"       
           #监控脚本检测周期秒
           interval 2    
           #是如果script中的指令执行失败，vrrp_instance的优先级会减少10个点。      
           weight  -10         
   }
vrrp_instance VI_1 {
    #backup机器配置为BACKUP
    state MASTER 
    #绑定的网卡名称
    interface ens32     
    #必须与backup机器配置相同
    virtual_router_id  51  
    #优先级，要比backup机器高至少50
    priority 100  
    #检查间隔，默认1秒 VRRP心跳包的发送周期，单位为s
    advert_int 1 
    authentication {
        #授权类型，必须与backup相同
        auth_type PASS  
        #授权密码，必须与backup相同
        auth_pass 1111   
    }
    virtual_ipaddress {
        #虚拟ip地址，xxx部分必须与nginx网段相同，必须与backup相同
        192.168.xxx.250   
    }
    track_script {
      chk_http_port
    }
}

4.backup配置

#vim /etc/keepalived/keepalived.conf
#复制下面内容：
global_defs {
   notification_email {
     test@qq.com
   }
   notification_email_from test@qq.com
   smtp_server 127.0.0.1
   smtp_connect_timeout 30
   router_id LVS_DEVEL_BAK
}
  vrrp_script chk_http_port {
           script "/etc/keepalived/checknginx.sh"       
           interval 2                                         
           weight -10 
        }
vrrp_instance VI_1 {
    state BACKUP
    interface ens32
    virtual_router_id 51
    priority 50
    advert_int 1
     authentication {
        auth_type PASS  
        auth_pass 1111   
    } 
    virtual_ipaddress {
        192.168.xxx.250   
    }
   track_script {
        chk_http_port
    } 
    #不抢占IP，即当keepalived发生切换后，只能配置在BACKUP上 ，
    #这样避免来回切换带来的系统开销，同时也避免了IP切换带来的系统不稳定性
    nopreempt                   
}
